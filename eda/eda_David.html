# --- STEP 1: Geometry (Silent & Clean) ---
sf_use_s2(FALSE)
states_sf <- suppressMessages(suppressWarnings(
  st_as_sf(maps::map("state", plot = FALSE, fill = TRUE)) |>
    mutate(state_name = str_to_title(word(ID, 1, sep = ":"))) |>
    st_make_valid() |> 
    group_by(state_name) |> 
    summarise(geometry = st_union(geom), .groups = "drop") |>
    st_transform(4326)
))

# --- STEP 2: AI Rates (Summarized & Defensive) ---
ai_rates <- tech_2018_clean |>
  ungroup() |> 
  filter(tech_type == "artificial intelligence") |>
  mutate(naics_sector = as.character(naics_sector)) |>
  group_by(state, naics_sector) |>
  summarise(
    ai_adoption_rate = sum(n_firms[used_tech == "Used"], na.rm = TRUE) / 
      sum(n_firms, na.rm = TRUE),
    .groups = "drop"
  )

# --- STEP 3: Payroll (Matching by Cleaned Sector Names) ---
payroll_stats <- cs_panel_clean |>
  ungroup() |> 
  group_by(state, naics_sector) |>
  summarise(
    mean_payroll_per_emp = mean(payroll_per_emp, na.rm = TRUE), # Raw USD
    .groups = "drop"
  )

# --- STEP 4: Final Merge & Industry Filter ---
state_ai_spatial <- ai_rates |>
  left_join(payroll_stats, by = c("state", "naics_sector")) |>
  left_join(fips_to_name, by = c("state" = "state_fips")) |>
  filter(!is.na(state_name))

# Identify industries with enough data to show in the dropdown
valid_industries <- state_ai_spatial |>
  group_by(naics_sector) |>
  summarise(missing = sum(is.na(ai_adoption_rate)), .groups = "drop") |>
  filter(missing <= 10, !naics_sector %in% c("Unknown", "Total")) |>
  pull(naics_sector)

# --- STEP 5: Shiny App ---
ui <- fluidPage(
  titlePanel("US AI Adoption & Payroll by Sector (2018)"),
  sidebarLayout(
    sidebarPanel(
      selectInput("industry_choice", "Select Industry Sector:", 
                  choices = sort(valid_industries),
                  selected = "Professional Services"),
      hr(),
      helpText("Hover over states for metrics. Grey indicates insufficient data.")
    ),
    mainPanel(leafletOutput("map", height = "750px"))
  )
)

server <- function(input, output, session) {
  
  filtered_data <- reactive({
    req(input$industry_choice)
    state_ai_spatial |>
      filter(naics_sector == input$industry_choice) |>
      left_join(states_sf, by = "state_name") |>
      st_as_sf() |>
      filter(!st_is_empty(geometry))
  })
  
  output$map <- renderLeaflet({
    leaflet() |>
      addProviderTiles(providers$CartoDB.Positron) |>
      setView(lng = -98.58, lat = 39.82, zoom = 4)
  })
  
  observe({
    data <- filtered_data()
    req(nrow(data) > 0)
    
    pal <- colorNumeric("YlOrRd", 
                        domain = c(0, max(data$ai_adoption_rate * 100, na.rm = TRUE)), 
                        na.color = "#D3D3D3")
    
    labels <- sprintf(
      "<strong>%s</strong><br/>
       Sector: %s<br/>
       AI Adoption: <b>%s</b><br/>
       Payroll/Emp: <b>%s</b>",
      data$state_name, 
      data$naics_sector, 
      ifelse(is.na(data$ai_adoption_rate), 
             "Insufficient Data", 
             paste0(round(data$ai_adoption_rate * 100, 2), "%")),
      # Applied / 1000 fix here for raw USD to 'k' display
      ifelse(is.na(data$mean_payroll_per_emp), 
             "N/A", 
             paste0("$", formatC(data$mean_payroll_per_emp / 1000, format="f", digits=1, big.mark=","), "k"))
    ) |> lapply(htmltools::HTML)
    
    leafletProxy("map", data = data) |>
      clearShapes() |>
      addPolygons(
        fillColor = ~pal(ai_adoption_rate * 100),
        fillOpacity = 0.8, color = "white", weight = 1,
        highlightOptions = highlightOptions(weight = 3, color = "#666", bringToFront = TRUE),
        label = labels
      ) |>
      clearControls() |>
      addLegend("bottomright", pal = pal, values = ~ai_adoption_rate * 100,
                title = "AI Adoption (%)", labFormat = labelFormat(suffix = "%"),
                na.label = "No Data")
  })
}

shinyApp(ui, server)
